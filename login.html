const express = require("express"); const path = require("path"); const app = express(); const PORT = 3000; // Allow JSON app.use(express.json()); // Serve all frontend files (HTML, CSS, JS) app.use(express.static(__dirname)); // ✅ Default route — opens login page first app.get("/", (req, res) => { res.sendFile(path.join(__dirname, "login.html")); }); // ✅ Serve grocery page app.get("/home", (req, res) => { res.sendFile(path.join(__dirname, "index.html")); }); // ✅ Serve payment page app.get("/payment", (req, res) => { res.sendFile(path.join(__dirname, "payment.html")); }); // ✅ Start the server app.listen(PORT, () => { console.log(✅ SmartGrocer server running at: http://localhost:${PORT}); });
